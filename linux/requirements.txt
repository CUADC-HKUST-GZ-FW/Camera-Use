numpy>=1.19.0

# jetson-stats  # Jetson设备监控工具

# Linux系统依赖（通过包管理器安装）：
# Ubuntu/Debian (包括Jetson):
#   sudo apt-get update
#   sudo apt-get install python3-dev python3-pip
#   sudo apt-get install python3-opencv  # 如果系统未预装
#   sudo apt-get install build-essential cmake
#   sudo apt-get install libusb-1.0-0-dev
#   sudo apt-get install nvidia-l4t-camera  # Jetson特定
#   sudo apt-get install v4l-utils  # 视频设备工具

# CentOS/RHEL (不适用于Jetson):
#   sudo yum install python3-devel python3-pip
#   sudo yum install opencv-devel python3-opencv
#   sudo yum install gcc gcc-c++ make cmake

# 海康威视SDK依赖（需要单独安装）：
# 1. 下载ARM64版本的MVS SDK (for Jetson aarch64)
#    文件名示例: MVS-2.1.2-Linux-aarch64-*.tar.gz
# 2. 解压到 /opt/MVS/ 或 /usr/local/MVS/
# 3. 设置环境变量：
#    export MVCAM_COMMON_RUNENV=/opt/MVS/lib
#    export LD_LIBRARY_PATH=/opt/MVS/lib/aarch64:/opt/MVS/lib:$LD_LIBRARY_PATH
#    export PYTHONPATH=/opt/MVS/Samples/aarch64/Python:$PYTHONPATH

# Jetson性能优化提示：
# 1. 启用最大性能模式：sudo nvpmodel -m 0
# 2. 设置最大CPU频率：sudo jetson_clocks
# 3. 监控温度和功耗：sudo tegrastats
